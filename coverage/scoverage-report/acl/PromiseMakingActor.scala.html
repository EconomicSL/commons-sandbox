24 /** Trait defining behavior for an actor that trades [[edsl.commitments.Commitment Commitment]] objects. */
25 trait PromiseMakingActor {
26   this: Actor with ActorLogging =>
27 
28   /** The act of accepting a previously submitted [[Promise]] to perform
29     * actions specified in some [[edsl.commitments.Commitment Commitment]].
30     *
31     * @param promise a [[Promise]] containing a [[edsl.commitments.Commitment Commitment]] specifying certain actions
32     *                that the `PromiseMakingActor` is to perform.
33     * @note `acceptPromise` is a general-purpose acceptance of a [[Promise]]
34     *       that was previously submitted (typically via a `makePromise` act).
35     *       The `PromiseMakingActor` sending [[PromiseAccepted]] is informing the
36     *       receiver that it intends to perform the actions as specified in
37     *       the [[edsl.commitments.Commitment Commitment]].
38     */
39   def acceptPromise(promise: Promise): Unit = {
40     promise.sender ! PromiseAccepted(self, promise.receiver, promise)
41   }
42 
43   /** The action of submitting a [[Promise]] made by the `PromiseMakingActor` to a
44     * collection of other `PromiseMakingActors` to perform actions specified in the
45     * [[edsl.commitments.Commitment Commitment]].
46     *
47     * @param receiver a collection of `PromiseMakingActors` who are receiving the
48     *                 [[Promise]] and therefore are potential counter parties to
49     *                 the [[edsl.commitments.Commitment Commitment]].
50     * @param contract a [[edsl.commitments.Commitment Commitment]] specifying certain actions that the various
51     *                 parties should perform.
52     * @note `makePromise` is a general-purpose act to make a [[Promise]] or
53     *       respond to a previously submitted [[Promise]] (i.e., during a
54     *       negotiation process) by promising to perform certain actions
55     *       defined by the [[edsl.commitments.Commitment Commitment]].
56     */
57   def makePromise(receiver: immutable.Set[ActorRef], contract: Commitment): Unit = {
58     receiver.foreach(r => r ! Promise(self, receiver, contract))
59   }
60 
61   /** The action of rejecting a previously submitted [[Promise]] to perform
62     * actions specified in some [[edsl.commitments.Commitment Commitment]].
63     *
64     * @param promise a [[Promise]] containing a [[edsl.commitments.Commitment Commitment]] specifying certain
65     *                actions that the `PromiseMakingActor` is to perform.
66     * @note `rejectPromise` is a general-purpose rejection to a [[Promise]]
67     *       that was previously submitted (typically via a `makePromise` act).
68     *       The `PromiseMakingActor` sending [[PromiseRejected]] is informing the
69     *       receiver that it has no intention to perform the actions as
70     *       specified in the promised [[edsl.commitments.Commitment Commitment]].
71     */
72   def rejectPromise(promise: Promise): Unit = {
73     promise.sender ! PromiseRejected(self, promise.receiver, promise)
74   }
75 }

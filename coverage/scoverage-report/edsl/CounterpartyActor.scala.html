23 /** Trait defining behavior for an actor that is a counterparty to the
24   * underlying [[edsl.commitments.Contract Contract]] of some
25   * [[edsl.ContractActorLike ContractActorLike]].
26   *
27   * @note A `CounterpartyActor` communicates with another `CounterpartyActor`
28   *       using a [[ContractActorLike ContractActorLike]] as a communication
29   *       "channel."
30   */
31 trait CounterpartyActor {
32   this: Actor with ActorLogging =>
33 
34   /** The balance sheet contains the `CounterpartyActor`'s collection of assets
35     * and liabilities.
36     */
37   def balanceSheet: BalanceSheetLike
38 
39   /** The act of breaking the terms of some existing [[edsl.commitments.Contract Contract]].
40     *
41     * @param receiver A [[edsl.ContractActorLike]] actor whose underlying
42     *                 [[edsl.commitments.Contract Contract]] the `Counterparty`
43     *                 actor wishes to break.
44     * @note When a `Counterparty` actor breaks the terms of some existing
45     *       [[edsl.commitments.Contract Contract]] it becomes in "breach
46     *       of contract." The `Counterparty` actor sending [[ContractBroken]]
47     *       is informing the receiver that it no longer has the intention of
48     *       performing the actions specified in the receiver's underlying
49     *       [[edsl.commitments.Contract Contract]].
50     */
51   def breakContract(receiver: ActorRef): Unit = {
52     receiver ! ContractBroken(self, receiver)
53   }
54 
55   /** Compute the value of the `Counterparty` actor's stock of assets.
56     *
57     * @param f A function mapping a [[ContractActorLike]] actor to a
58     *          [[edsl.commitments.Contract Contract]] representing the value
59     *          of the [[ContractActorLike]] actor's underlying
60     *          [[edsl.commitments.Contract Contract]] to the `Counterparty` actor.
61     * @return The total value of the `Counterparty` actor's stock of assets.
62     */
63   def valueAssets(f: ActorRef => Contract): Contract = {
64     balanceSheet.assets.map(f).reduce((contract1, contract2) => And(contract1, contract2))
65   }
66 
67   /** Compute the value of the `Counterparty` actor's equity.
68     *
69     * @param f A function mapping a [[ContractActorLike]] actor to a
70     *          [[edsl.commitments.Contract Contract]] representing the value
71     *          of the [[ContractActorLike]] actor's underlying
72     *          [[edsl.commitments.Contract Contract]] to the `Counterparty` actor.
73     * @return The total value of the `Counterparty` actor's stock of assets.
74     * @note By definition, the value of equity is the difference between the
75     *       value of the `Counterparty` actor's stock of assets and the value
76     *       of the `Counterparty` actor's stock of liabilities.
77     */
78   def valueEquity(f: ActorRef => Contract): Contract = {
79     And(valueAssets(f), Give(valueLiabilities(f)))
80   }
81 
82   /** Compute the value of the `Counterparty` actor's stock of liabilities.
83     *
84     * @param f A function mapping a [[ContractActorLike]] actor to a
85     *          [[edsl.commitments.Contract Contract]] representing the value
86     *          of the [[ContractActorLike]] actor's underlying
87     *          [[edsl.commitments.Contract Contract]] to the `Counterparty` actor.
88     * @return The total value of the `Counterparty` actor's stock of liabilities.
89     */
90   def valueLiabilities(f: ActorRef => Contract): Contract = {
91     balanceSheet.liabilities.map(f).reduce((contract1, contract2) => And(contract1, contract2))
92   }
93 
94   /** Defines the behavior of a `Counterparty` actor. */
95   def counterpartyBehavior: Receive = {
96     case AddAsset(contract) => balanceSheet.assets.add(contract)
97     case AddLiability(contract) => balanceSheet.liabilities.add(contract)
98     case RemoveAsset(contract) => balanceSheet.assets.remove(contract)
99     case RemoveLiability(contract) => balanceSheet.liabilities.remove(contract)
100   }
101 
102 }

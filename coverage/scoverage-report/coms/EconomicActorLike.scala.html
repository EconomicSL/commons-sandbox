24 /** Base trait for all economic actors. */
25 trait EconomicActorLike {
26   this: Actor with ActorLogging =>
27 
28   /** Balance sheet. */
29   def balanceSheet: ActorRef
30 
31   /** The action of accepting a previously submitted proposal to perform actions
32     * specified in some contract.
33     * @param proposal a proposal containing a contract specifying certain
34     *                 actions that the actor is to perform.
35     * @note `acceptProposal` is a general-purpose acceptance of a proposal that
36     *       was previously submitted (typically via a propose act). The agent
37     *       sending `ProposalAccepted` is informing the receiver that it intends
38     *       to perform the actions as specified in the proposal's contract.
39     */
40   def acceptProposal(proposal: Proposal): Unit = {
41     proposal.sender ! ProposalAccepted(self, proposal.receiver, proposal)
42   }
43 
44   /** The action of submitting a proposal made by the actor to a collection of
45     * other actors to perform actions specified in the contract.
46     * @param receiver a collection of actors who are receiving the proposal and
47     *                 therefore are counter party to the contract.
48     * @param contract a contract specifying certain actions that the various
49     *                 parties should perform.
50     * @note propose is a general-purpose act to make a proposal or respond to
51     *       an existing proposal (possibly during a negotiation process) by
52     *       proposing to perform certain actions defined by the contract.
53     */
54   def propose(receiver: immutable.Set[ActorRef], contract: Contract): Unit = {
55     receiver.foreach(r => r ! Proposal(self, receiver, contract))
56   }
57 
58   /** The action of rejecting a previously submitted proposal to perform actions
59     * specified in some contract.
60     * @param proposal a proposal containing a contract specifying certain
61     *                 actions that the actor is to perform.
62     * @note `rejectProposal` is a general-purpose rejection to a proposal that
63     *       was previously submitted (typically via a propose act). The agent
64     *       sending `ProposalRejected` is informing the receiver that it has no
65     *       intention to perform the actions as specified in the proposal's
66     *       contract.
67     */
68   def rejectProposal(proposal: Proposal): Unit = {
69     proposal.sender ! ProposalRejected(self, proposal.receiver, proposal)
70   }
71 }

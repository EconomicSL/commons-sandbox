26 /** Trait representing a generic economic actor. */
27 trait EconomicActorLike {
28   this: Actor with ActorLogging =>
29 
30   /** A unique identifier. */
31   def id: UUID
32 
33   /** Balance sheet. */
34   def balanceSheet: ActorRef
35 
36   /** The action of accepting a previously submitted proposal to perform actions
37     * specified in some contract.
38     * @param proposal a proposal containing a contract specifying certain
39     *                 actions that the actor is to perform.
40     * @note `acceptProposal` is a general-purpose acceptance of a proposal that
41     *       was previously submitted (typically via a propose act). The agent
42     *       sending `ProposalAccepted` is informing the receiver that it intends
43     *       to perform the actions as specified in the proposal's contract.
44     */
45   def acceptProposal(proposal: Proposal): Unit = {
46     proposal.sender ! ProposalAccepted(self, proposal.receiver, proposal)
47   }
48 
49   /** The action of submitting a proposal made by the actor to a collection of
50     * other actors to perform actions specified in the contract.
51     * @param receiver a collection of actors who are receiving the proposal and
52     *                 therefore are counter party to the contract.
53     * @param contract a contract specifying certain actions that the various
54     *                 parties should perform.
55     * @note propose is a general-purpose act to make a proposal or respond to
56     *       an existing proposal (possibly during a negotiation process) by
57     *       proposing to perform certain actions defined by the contract.
58     */
59   def propose(receiver: immutable.Set[ActorRef], contract: Contract): Unit
60 
61   /** The action of rejecting a previously submitted proposal to perform actions
62     * specified in some contract.
63     * @param proposal a proposal containing a contract specifying certain
64     *                 actions that the actor is to perform.
65     * @note `rejectProposal` is a general-purpose rejection to a proposal that
66     *       was previously submitted (typically via a propose act). The agent
67     *       sending `ProposalRejected` is informing the receiver that it has no
68     *       intention to perform the actions as specified in the proposal's
69     *       contract.
70     */
71   def rejectProposal(proposal: Proposal): Unit = {
72     proposal.sender ! ProposalRejected(self, proposal.receiver, proposal)
73   }
74 }

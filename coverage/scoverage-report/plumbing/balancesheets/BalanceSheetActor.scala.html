24 /** Class representing a balance sheet.
25   * @note A [[BalanceSheetActor]] should be a child of some [[EconomicActorLike]] actor.
26   */
27 abstract class BalanceSheetActor extends Actor
28   with ActorLogging {
29 
30   /** Unordered collection of tradable objects that are assets. */
31   def assets: mutable.Set[ActorRef]
32 
33   /** Unordered collection of tradable objects that are liabilities. */
34   def liabilities: mutable.Set[ActorRef]
35 
36   def receive: Receive = {
37     case AddAsset(asset) =>
38       assets.add(asset)
39     case RemoveAsset(asset) =>
40       assets.remove(asset)
41     case AddLiability(liability) =>
42       liabilities.add(liability)
43     case RemoveLiability(liability) =>
44       liabilities.remove(liability)
45 
46   }
47 }

23 /** Class representing a balance sheet.
24   * @note A [[BalanceSheetActor]] should be a child of some [[actor.EconomicActor]].
25   */
26 abstract class BalanceSheetActor extends Actor
27   with ActorLogging {
28 
29   /** Unordered collection of tradable objects that are assets. */
30   def assets: mutable.Set[ActorRef]
31 
32   /** Unordered collection of tradable objects that are liabilities. */
33   def liabilities: mutable.Set[ActorRef]
34 
35   def receive: Receive = {
36     case AddAsset(asset) =>
37       assets.add(asset)
38     case RemoveAsset(asset) =>
39       assets.remove(asset)
40     case AddLiability(liability) =>
41       liabilities.add(liability)
42     case RemoveLiability(liability) =>
43       liabilities.remove(liability)
44 
45   }
46 }

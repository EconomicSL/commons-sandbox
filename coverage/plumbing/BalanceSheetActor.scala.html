24 /** Class representing a financial balance sheet.
25   *
26   * @note A [[BalanceSheetActor]] should be a child of some [[actor.EconomicActor]].
27   */
28 abstract class BalanceSheetActor extends Actor
29   with ActorLogging {
30 
31   /** Unordered collection of tradable objects. */
32   def contracts: mutable.Set[Tradable]
33 
34   /** Unordered collection of tradable object that are assets. */
35   def assets: mutable.Set[Tradable] = {
36     contracts.filter {
37       case tradable: Tangible => true
38       case tradable: ContractLike if tradable.promisee == context.parent => true
39       case _ => false
40     }
41   }
42 
43   /** Unordered collection of tradable object that are liabilities. */
44   def liabilities: mutable.Set[Tradable] = {
45     contracts.filter {
46       case tradable: ContractLike if tradable.promisor == context.parent => true
47       case _ => false
48     }
49   }
50 
51   def receive: Receive = {
52     case PutTradable(tradable) =>
53       contracts.add(tradable)
54     case DeleteTradable(tradable) =>
55       contracts.remove(tradable)
56 
57   }
